10   def add(inputs: list[numbers.Number]) -> numbers.Number:
11       total: numbers.Number = 0
12       for number in inputs:
13           total += number
14   
15       return total

18   def subtract(inputs: list[numbers.Number]) -> numbers.Number:
19       total: numbers.Number = inputs[0]
20       for i in range(1,len(inputs)):
21           total -= inputs[i]
22   
23       return total

26   def multiply(inputs: list[numbers.Number]) -> numbers.Number:
27       if 0 in inputs:
28           return 0
29       total: numbers.Number = 1
30       for number in inputs:
31           total *= number
32   
33       return total

35   
36   def divide(inputs: list[numbers.Number]) -> numbers.Number:
37       divisors = inputs[1:]
38       if 0 in divisors:
39           raise ValueError("Input list of numbers included zero. Can't divide by zero.")
40       if inputs[0] == 0:
41           return 0
42   
43       total: numbers.Number = inputs[0]
44       for i in range(1,len(inputs)):
45           total /= inputs[i]
46   
47       return total
